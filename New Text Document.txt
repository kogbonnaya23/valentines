  <script>
    const yesBtn = document.getElementById("yesBtn");
    const noBtn = document.getElementById("noBtn");
    const teddyOverlay = document.getElementById("teddyOverlay");

    // Make the "No" button run away
    function moveNoButton(event) {
      const rect = noBtn.getBoundingClientRect();
      const btnCenterX = rect.left + rect.width / 2;
      const btnCenterY = rect.top + rect.height / 2;

      const offsetX = event.clientX - btnCenterX;
      const offsetY = event.clientY - btnCenterY;

      const distance = Math.sqrt(offsetX ** 2 + offsetY ** 2);
      const threshold = 110;

      if (distance < threshold) {
        const padding = 40;
        const newLeft = padding + Math.random() * (window.innerWidth - rect.width - padding * 2);
        const newTop = padding + Math.random() * (window.innerHeight - rect.height - padding * 2);

        noBtn.style.left = newLeft + "px";
        noBtn.style.top = newTop + "px";
        noBtn.style.transform = "translate(0, 0)";
      }
    }

    document.addEventListener("mousemove", moveNoButton);

    noBtn.addEventListener("click", (e) => {
      e.preventDefault();
      const padding = 40;
      const rect = noBtn.getBoundingClientRect();
      const newLeft = padding + Math.random() * (window.innerWidth - rect.width - padding * 2);
      const newTop = padding + Math.random() * (window.innerHeight - rect.height - padding * 2);
      noBtn.style.left = newLeft + "px";
      noBtn.style.top = newTop + "px";
    });

    // Show teddy when "Yes" is clicked
    yesBtn.addEventListener("click", () => {
      teddyOverlay.classList.add("show");
    });

    // Hide teddy when overlay is clicked
    teddyOverlay.addEventListener("click", () => {
      teddyOverlay.classList.remove("show");
    });

    // Floating hearts background
    const HEART_COUNT = 16;

    function createFloatingHeart() {
      const heart = document.createElement("div");
      heart.className = "floating-heart";
      const startLeft = Math.random() * 100;
      const delay = Math.random() * 10;
      const scale = 0.5 + Math.random() * 0.9;
      heart.style.left = startLeft + "vw";
      heart.style.animationDelay = delay + "s";
      heart.style.transform = `scale(${scale}) rotate(-45deg)`;
      document.body.appendChild(heart);

      heart.addEventListener("animationend", () => {
        heart.remove();
        createFloatingHeart();
      });
    }

    for (let i = 0; i < HEART_COUNT; i++) {
      createFloatingHeart();
    }
  </script>
